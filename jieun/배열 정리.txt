#배열
하나의 이름을 갖고, 동일한 자료형의 변수를 여러 개 생성!
동일한 자료형이기때문에 크기가 전부 같음

-선언방식 
int scores[10];
(자료형-배열이름-요소개수)

-성적 처리 예제
#include <stdio.h>
#define STUDENTS 10
int main()
{
	int score[STUDENTS];
	int sum = 0;
	int i, average;
	for (i = 0; i < STUDENTS; i++)
	{
		printf("Enter score: ");
		scanf("%d", &score[i]);
	}
	for (i = 0; i < STUDENTS; i++)
		sum += score[i];
	average = sum / STUDENTS;
	printf("Average = %d\n", average);
	return 0;
}

#배열의 복사
int src[SIZE];
int dst[SIZE];
// 잘못된 배열 복사, 컴파일 오류!
dst = src;
// 올바른 배열 복사, 모든 원소를 복사해야 한다.
for (i = 0; i < SIZE; i++)
dst[i] = src[i];

함수를 통해서 원본 배열이 변경될 수 있음
#include <stdio.h>
#define SIZE 7

void print_array(int a[], int size)
{
	for (int i = 0; i < size; i++)
		printf("%2d ", a[i]);
	printf("\n");
}

void square_array(int a[], int size)
{
	for (int i = 0; i < size; i++)
		a[i] = a[i] * a[i];
}

int main(void)
{
	int list[SIZE] = { 1,2,3,4,5,6,7 };
	print_array(list, SIZE);
	square_array(list, SIZE);
	print_array(list, SIZE);
	return 0;
}
