두문장이상일 때는 조건문 처럼 중괄호가 필요함

#예제: 1~n 합 구하기
#include <stdio.h>

int main(void)
{
	int i, n, sum;
	printf("Enter number: ");
	scanf("%d", &n);
	i = 1;
	sum = 0;
	while (i <= n){	
		sum += i;
		i++;
	}
	printf("Sum of 1 to %d is %d\n", n, sum);
	return 0;

}

//원금의 두배가 되기까지 도달하는 시간

#include <stdio.h>
#define SEED_MONEY 100000000

int main(void){
	int year = 0;
	int money = SEED_MONEY;
	float rate = 0.0;
	printf("Annual interest rate?: ");
	scanf("%f", &rate);
	while (money < SEED_MONEY * 2){
		money += money * rate;
		year ++;
	
	}
	printf("%d year", year);
	return 0;
}


#include <stdio.h>
#define SEED_MONEY 100000000

int get_factorial(int n){
	for (int i= n-1; i > 0; i--){
		n = n * i;
	}
	return n;
}
int main(void){
	int n;
	printf("Enter number: ");
	scanf("%d", &n);
	printf("%d! is %d\n", n, get_factorial(n));
	return 0;
}
